-- FARM MODULE
CREATE TABLE FARM
(
    FARM_ID INTEGER DEFAULT FARM_SEQ.NEXTVAL,
    FARM_NAME VARCHAR2(30),
    FARM_ADDRESS VARCHAR2(60),
    FARM_PHONE CHAR(12),
    FARM_SIZE NUMBER(5,2),
    FARM_IMG BLOB,

    CONSTRAINT FARM_PK PRIMARY KEY(FARM_ID),
    CONSTRAINT FARM_NAME_UQ UNIQUE(FARM_NAME)
);

CREATE TABLE STOCKYARD
(
    STOCKYARD_ID INTEGER DEFAULT STOCKYARD_SEQ.NEXTVAL,
    STOCKYARD_LABEL VARCHAR2(20),
    FARM_ID INTEGER,
    STOCKYARD_IMG BLOB,

    CONSTRAINT STOCKYARD_PK PRIMARY KEY(STOCKYARD_ID),
    CONSTRAINT FARM_STOCKYARD_FK FOREIGN KEY(FARM_ID) REFERENCES FARM(FARM_ID) ON DELETE CASCADE
);

CREATE TABLE BREED
(
    BREED_ID INTEGER DEFAULT BREED_SEQ.NEXTVAL,
    BREED_NAME VARCHAR2(30),
    BREED_DESCRIPTION VARCHAR2(80),
    REGION_OF_ORIGIN VARCHAR2(30),
    BREED_IMG BLOB,


    CONSTRAINT BREED_PK PRIMARY KEY(BREED_ID)
);

CREATE TABLE COW
(
    COW_ID INTEGER DEFAULT COW_SEQ.NEXTVAL,
    STOCKYARD_ID INTEGER,
    BREED_ID INTEGER,
    COW_CONTROLNUM CHAR(6) NOT NULL,
    -- COW_AGE NUMBER(2,0),
    COW_BIRTH_DATE DATE NOT NULL,
    COW_GENDER CHAR(1) NOT NULL,

    CONSTRAINT COW_PK PRIMARY KEY(COW_ID),
    CONSTRAINT STOCKYARD_FK FOREIGN KEY(STOCKYARD_ID) REFERENCES STOCKYARD(STOCKYARD_ID) ON DELETE CASCADE,
    CONSTRAINT BREED_FK FOREIGN KEY(BREED_ID) REFERENCES BREED(BREED_ID) ON DELETE CASCADE,
    CONSTRAINT COW_GENDER_CK CHECK (COW_GENDER IN ('M', 'F'))
);

CREATE TABLE CUSTOMER
(
    CUSTOMER_ID INTEGER DEFAULT CUSTOMER_SEQ.NEXTVAL,
    FNAME VARCHAR2(50) NOT NULL,
    LNAME VARCHAR2(50) NOT NULL,
    NRIC CHAR(12) NOT NULL,
    PHONE CHAR(12) NOT NULL,
    EMAIL VARCHAR2(100) NOT NULL,
    PWD VARCHAR(20) NOT NULL,

    CONSTRAINT CUSTOMER_PK PRIMARY KEY(CUSTOMER_ID),
    CONSTRAINT FNAME_UQ UNIQUE(FNAME),
    CONSTRAINT LNAME_UQ UNIQUE(LNAME)

CREATE TABLE DELIVERY_ADDRESS
(
    DELIVERY_ADDRESS_ID INTEGER DEFAULT DELIVERY_ADDRESS_SEQ.NEXTVAL,
    CUSTOMER_ID INTEGER,
    STREET VARCHAR2(60) NOT NULL,
    STATE VARCHAR2(30) NOT NULL,
    AREA VARCHAR2(30) NOT NULL,
    POSTAL_CODE CHAR(5) NOT NULL,

    CONSTRAINT DELIVERY_ADDRESS_PK PRIMARY KEY (DELIVERY_ADDRESS_ID),
    CONSTRAINT CUSTOMER_FK FOREIGN KEY (CUSTOMER_ID) REFERENCES CUSTOMER(CUSTOMER_ID) ON DELETE CASCADE
);

CREATE TABLE ORDERS
(
    ORDERS_ID INTEGER DEFAULT ORDERS_SEQ.NEXTVAL,
    DELIVERY_ADDRESS_ID INTEGER,
    ORDER_DATE DATE NOT NULL,
    TOTAL_PRICE NUMBER(5, 2) NOT NULL,
    PAYMENT_METHOD VARCHAR2(30) NOT NULL,
    ISPAID NUMBER(1, 0) DEFAULT 0,

    CONSTRAINT ORDERS_PK PRIMARY KEY (ORDERS_ID),
    CONSTRAINT DELIVERY_ADDRESS_FK FOREIGN KEY (DELIVERY_ADDRESS_ID) REFERENCES DELIVERY_ADDRESS(DELIVERY_ADDRESS_ID) ON DELETE CASCADE,
    CONSTRAINT PAYMENT_METHOD_CK CHECK (PAYMENT_METHOD IN ('Bank Transfer', 'Credit/debit Card')),
    CONSTRAINT ISPAID_CK CHECK (ISPAID IN (0, 1))
);

CREATE TABLE ORDER_PRODUCT
(
    ORDER_PROD_ID INTEGER DEFAULT ORDER_PRODUCT_SEQ.NEXTVAL,
    ORDERS_ID INTEGER,
    DAIRY_PRODUCT_ID INTEGER,
    SUBTOTAL NUMBER(5, 2) NOT NULL,
    QUANTITY NUMBER(2, 0) NOT NULL,

    CONSTRAINT ORDER_PRODUCT_PK PRIMARY KEY(ORDER_PROD_ID),
    CONSTRAINT ORDERS_FK FOREIGN KEY (ORDERS_ID) REFERENCES ORDERS(ORDERS_ID) ON DELETE CASCADE,
    CONSTRAINT ORDER_PRODUCT_FK FOREIGN KEY (DAIRY_PRODUCT_ID) REFERENCES DAIRY_PRODUCT(DAIRY_PRODUCT_ID) ON DELETE CASCADE
);

CREATE TABLE CARD

    CONSTRAINT CARD_PK PRIMARY KEY (ORDERS_ID),
    CONSTRAINT ORDERS_CARD_FK FOREIGN KEY (ORDERS_ID) REFERENCES ORDERS(ORDERS_ID) ON DELETE CASCADE
